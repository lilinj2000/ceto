TEST(JsonTest, escapeTest) {
  Document config;
  soil::json::load_from_file(&config, "log.json");
  soil::log::init(config);

  std::string test("123 \0 456", 9);
  StringBuffer s;
  Writer<StringBuffer> writer(s);
  writer.StartObject();
  writer.Key("test");
  writer.String(test.data(), test.length());
  writer.EndObject();
  SOIL_DEBUG_PRINT(s.GetString());

  Document doc;
  if (doc.Parse(s.GetString()).HasParseError()) {
    SOIL_DEBUG_PRINT(soil::json::get_parse_error(doc, test));
    FAIL();
  }
  std::string value;
  soil::json::get_item_value(&value, doc, "/test");

  EXPECT_EQ(9, value.length());
  EXPECT_EQ(value, test);

  std::string esp_test = soil::json::escape_string(test);
  SOIL_DEBUG_PRINT(esp_test);
  EXPECT_EQ("123 \\u0000 456", esp_test);
}
